cat > /workspace/entrypoint.sh << 'SCRIPT'
#!/bin/bash
# /workspace/entrypoint.sh — Autonomous training entrypoint for Akash
# NOTE: Do NOT use "set -e" — pgrep returns non-zero normally, which would kill the script.

LOG_DIR="/workspace/logs"
mkdir -p "$LOG_DIR" /workspace/data /workspace/checkpoints /workspace/scripts /workspace/cache/huggingface /workspace/cache/pip /run/sshd

echo "[$(date)] Entrypoint started" >>"$LOG_DIR/entrypoint.log"

apt-get update -qq && apt-get install -y -qq openssh-server git wget >/dev/null 2>&1 || true
apt-get clean && rm -rf /var/lib/apt/lists/* 2>/dev/null || true

echo "root:${SSH_PASSWORD:-changeme}" | chpasswd
sed -i 's/#PermitRootLogin.*/PermitRootLogin yes/' /etc/ssh/sshd_config
sed -i 's/#PasswordAuthentication.*/PasswordAuthentication yes/' /etc/ssh/sshd_config
/usr/sbin/sshd || true
echo "[$(date)] SSH started" >>"$LOG_DIR/entrypoint.log"

pip install -q --cache-dir=/workspace/cache/pip unsloth trl datasets wandb typer rich datasketch huggingface_hub jupyterlab 2>>"$LOG_DIR/pip.log" || true
echo "[$(date)] Pip packages installed" >>"$LOG_DIR/entrypoint.log"

nvidia-smi >>"$LOG_DIR/entrypoint.log" 2>&1 || true

nohup jupyter lab --ip=0.0.0.0 --port=8888 --no-browser --allow-root --NotebookApp.token='' --NotebookApp.password='' --notebook-dir=/workspace >"$LOG_DIR/jupyter.log" 2>&1 &
echo "[$(date)] Jupyter started (PID: $!)" >>"$LOG_DIR/entrypoint.log"

echo "=== SLM Training Container Ready ==="

while [ -f /workspace/.auto-start-training ]; do
    if ! pgrep -f "train_cpt.py" >/dev/null 2>&1; then
        echo "[$(date)] Training not running. Checking for checkpoints..." >>"$LOG_DIR/entrypoint.log"
        LATEST_CKPT=$(ls -td /workspace/checkpoints/cpt/checkpoint-* 2>/dev/null | head -1)
        if [ -n "$LATEST_CKPT" ]; then
            echo "[$(date)] Resuming from $LATEST_CKPT" >>"$LOG_DIR/entrypoint.log"
            nohup python /workspace/scripts/train_cpt.py --resume-from "$LATEST_CKPT" >>"$LOG_DIR/cpt_training.log" 2>&1 &
            echo "[$(date)] Training resumed (PID: $!)" >>"$LOG_DIR/entrypoint.log"
        else
            echo "[$(date)] No checkpoint found. Starting fresh." >>"$LOG_DIR/entrypoint.log"
            nohup python /workspace/scripts/train_cpt.py >>"$LOG_DIR/cpt_training.log" 2>&1 &
            echo "[$(date)] Training started fresh (PID: $!)" >>"$LOG_DIR/entrypoint.log"
        fi
    fi
    sleep 300
done

echo "[$(date)] Flag file removed or absent. Keeping container alive." >>"$LOG_DIR/entrypoint.log"
while true; do sleep 86400; done
SCRIPT
chmod +x /workspace/entrypoint.sh
